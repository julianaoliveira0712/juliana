<script>
    function calcular(){
        //recuperando valores de campo como NUMEROS
        var numero1 = Number(document.getElementById("numero1").value);
        var numero2 = Number(document.getElementById("numero2").value);

        var potencia = numero1 ** numero2;

        var raiz1 = Math.sqrt(numero1);
        var raiz2 = Math.sqrt(numero2);

        document.getElementById("potencia").innerHTML = potencia;
        document.getElementById("raiz1").innerHTML = raiz1;
        document.getElementById("raiz2").innerHTML = raiz2;


    }
